CREATE OR REPLACE PROCEDURE PROC06 (
 i_num IN number
)
iS

BEGIN

FOR C1 in (SELECT hitcnt from article where ANUM=i_num) LOOP
    update article set hitcnt=c1.hitcnt+1 where anum=i_num;
  END LOOP;

END PROC06;
